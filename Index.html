let blockSize = 25; let total_row = 17; let total_col = 17; let board; let context; let snakeX = blockSize * 5; let snakeY = blockSize * 5; let speedX = 0; let speedY = 0; let snakeBody = []; let foodX; let foodY; let gameOver = false; let touchStartX, touchStartY, touchEndX, touchEndY; window.onload = function() { board = document.getElementById("board"); board.height = total_row * blockSize; board.width = total_col * blockSize; context = board.getContext("2d"); placeFood(); document.addEventListener("touchstart", touchStart); document.addEventListener("touchend", touchEnd); setInterval(update, 1000 / 10); } function touchStart(event) { touchStartX = event.touches[0].pageX; touchStartY = event.touches[0].pageY; } function touchEnd(event) { touchEndX = event.changedTouches[0].pageX; touchEndY = event.changedTouches[0].pageY; let deltaX = touchEndX - touchStartX; let deltaY = touchEndY - touchStartY; if (Math.abs(deltaX) > Math.abs(deltaY)) { if (deltaX > 0) { speedX = 1; speedY = 0; } else { speedX = -1; speedY = 0; } } else { if (deltaY > 0) { speedX = 0; speedY = 1; } else { speedX = 0; speedY = -1; } } } function placeFood() { foodX = Math.floor(Math.random() * total_col) * blockSize; foodY = Math.floor(Math.random() * total_row) * blockSize; } function update() { if (gameOver) { return; } context.fillStyle = "green"; context.fillRect(0, 0, board.width, board.height); context.fillStyle = "yellow"; context.fillRect(foodX, foodY, blockSize, blockSize); if (snakeX == foodX && snakeY == foodY) { snakeBody.push([foodX, foodY]); placeFood(); } for (let i = snakeBody.length - 1; i > 0; i--) { snakeBody[i] = snakeBody[i - 1]; } if (snakeBody.length) { snakeBody[0] = [snakeX, snakeY]; } snakeX += speedX * blockSize; snakeY += speedY * blockSize; if (snakeX < 0 || snakeX >= board.width || snakeY < 0 || snakeY >= board.height) { gameOver = true; } for (let i = 0; i < snakeBody.length; i++) { if (snakeX == snakeBody[i][0] && snakeY == snakeBody[i][1]) { gameOver = true; } } context.fillStyle = "red"; for (let i = 0; i < snakeBody.length; i++) { context.fillRect(snakeBody[i][0], snakeBody[i][1], blockSize, blockSize); } context.fillRect(snakeX, snakeY, blockSize, blockSize); }
